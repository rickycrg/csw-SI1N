import pandas as pd

    # Exemplo de dados
    dados = {
        'valor': [10, 12, 13, 14, 15, 100, 16, 17, 18, 19, 20]
    }
    df = pd.DataFrame(dados)
    
    # Calculando Q1, Q3 e IQR
    Q1 = df['valor'].quantile(0.25)
    Q3 = df['valor'].quantile(0.75)
    IQR = Q3 - Q1
    
    # Definindo os limites para os outliers
    limite_inferior = Q1 - 1.5 * IQR
    limite_superior = Q3 + 1.5 * IQR
    
    # Filtrando os dados, removendo os outliers
    df_sem_outliers = df[(df['valor'] >= limite_inferior) & (df['valor'] <= limite_superior)]
    
    print(df_sem_outliers)
